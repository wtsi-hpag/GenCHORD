\doxysubsection{Detailed Description}
\subsubsection*{template$<$\mbox{\hyperlink{structJSL_1_1NameValuePair}{int}} Dimension = 1, \mbox{\hyperlink{structJSL_1_1NameValuePair}{bool}} Delete\+Mode = true, \mbox{\hyperlink{structJSL_1_1NameValuePair}{char}} Symbol = \textquotesingle{}\#\textquotesingle{}, \mbox{\hyperlink{structJSL_1_1NameValuePair}{unsigned}} \mbox{\hyperlink{structJSL_1_1NameValuePair}{int}} Max\+Hashes = 20$>$\newline
class JSL\+::\+Progress\+Bar$<$ Dimension, Delete\+Mode, Symbol, Max\+Hashes $>$}
A neat display widget for tracking the progress of a process by printing a progress bar on the screen, which updates to visually indicate the process is running. Supports arbitrary "{}levels"{} of progress bars which can be independently modified through a variadic argument interface. 
\begin{DoxyTemplParams}{Template Parameters}
{\em Dimension} & The number of concurrent bars to print (fixes the number of arguments which must be provided to subsequent functions). Defaults to 1 if 0 template arguments provided \\
\hline
{\em Delete\+Mode} & If true, utilises \doxylink{namespaceJSL_ae38780f94d60045b904b289eb3ab8eda}{JSL\+::delete\+Line()} to remove the previous line before reprinting, such that the bar appears animated. If false, provides a more limited form of animation (false only available in Dimension==1 mode). Defaults to true. \\
\hline
{\em Symbol} & The symbol which is printed within the bar to fill up space. Defaults to "{}\#"{}. \\
\hline
{\em Max\+Hashes} & The number of symbols in a full bar (dictates the width of the progress bar) \\
\hline
\end{DoxyTemplParams}
