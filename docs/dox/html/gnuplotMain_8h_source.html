138 void Show()
139 {
140 std::string gpFile = DirName + "/" + "plotter.gp";
141 InitialiseOutput(gpFile);
142 if (Axes.size() > 0 || Axes[0].size() > 0)
143 {
144 WriteMultiplotToFile(gpFile);
145 }
146 for (int i = 0; i < Axes.size(); ++i)
147 {
148 for (int j = 0; j < Axes[i].size(); ++j)
149 {
150 if (Axes[i][j].Lock == false)
151 writeStringToFile(gpFile, Axes[i][j].Show());
152 }
153 }
154
155 // try-catch loop means if gnuplot fails, can still cleanup after itself without leaving tempfiles hanging around
156 try
157 {
158 systemCall("gnuplot -p " + gpFile);
159 }
160 catch (const std::exception &e)
161 {
162 CleanupTempFiles();
163 Error(PlottingError, "The gnuplot command failed. Could not show plot.");
